Basic Form Code (Bootstrap 2.3.2)

Forms on IMH are not usually made in raw HTML.

{# HTML code should always be indented 4 spaces on IMH #}
<div class="well well-deep">
    <form action="#" method="post" class="form-vertical formname">
        <div class="row-fluid">
            <div class="span12">
                <div id="formname">
                    <div class="control-group">
                        <label class="control-label required" for="form_fullname">Full Name</label>
                        <div class="controls">
                            <input type="text" id="form_fullname" name="form[fullname]" required="required" class="span12" placeholder="Full Name*">
                        </div>
                    </div>
                    <div class="control-group">
                        <label class="control-label required" for="form_url">Website URL</label>
                        <div class="controls">
                            <input type="text" id="form_url" name="form[url]" required="required" class="span12" placeholder="Your Website URL*">
                        </div>
                    </div>
                    <div class="control-group">
                        <label class="control-label" for="form_overall_satisfaction">
                            What is your overall satisfaction with our web hosting service?
                        </label>
                        <div class="controls">
                            <select id="form_overall_satisfaction" name="form[overall_satisfaction]" class="span12">
                                <option value=""></option>
                                <option value="Yes">Yes - Very Satisfied</option>
                                <option value="No">No - Very Dissatisfied</option>
                            </select>
                        </div>
                    </div>
                    <div class="control-group">
                        <label class="control-label" for="form_additional_comments">
                            Additional Comments:
                        </label>
                        <div class="controls">
                            <textarea id="form_additional_comments" name="form[additional_comments]" class="span12"></textarea>
                        </div>
                    </div>
                </div>
            </div>
        </div>
        <small>Required fields are marked with an asterisk (*)</small>
        <div class="row-fluid">
            <div class="form-actions span7 offset5">
                <button type="submit" class="btn btn-primary form-submit">Send</button>
            </div>
        </div>
    </form>
</div>
